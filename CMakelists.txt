cmake_minimum_required(VERSION 3.13.0 FATAL_ERROR)

project(Mirage)

set(Mirage_VERSION_MAJOR 1)
set(Mirage_VERSION_MINOR 0)

add_library(Mirage SHARED)
add_executable(Game)

target_link_libraries(Mirage Opengl32.lib)

target_link_libraries(Game Mirage)
target_include_directories(Game PRIVATE mirage/src)

target_compile_definitions(Mirage PRIVATE BUID_DLL)
# PLATFORM DEFINITIONS AND SETTINGS
if(WIN32)
target_compile_definitions(Mirage PRIVATE PLATFORM_WINDOWS)
set(CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} /SUBSYSTEM:WINDOWS")
target_compile_definitions(Game PRIVATE PLATFORM_WINDOWS)
endif()

set_property(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT Game)

add_subdirectory(mirage)
add_subdirectory(sandbox)
